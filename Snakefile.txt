rule preprocess_samples:
    input:
        samp="samples/simplified/",
        meta="metadata/metadata.csv",
        config="configs/preprocessing_simple.txt"
    output:
        samp_out="preprocessing_output/processed_samples.csv",
        stat_out="results/processed_samples_stats.csv",
        term_count_unprocessed="results/unprocessed_samples_term_count.csv",
        term_count_processed="results/processed_samples_term_count.csv"
    shell:
        "python scripts/term_processing.py {input.samp} {input.meta} {output.samp_out} {output.stat_out} {output.term_count_unprocessed} {output.term_count_processed} {input.config}"
